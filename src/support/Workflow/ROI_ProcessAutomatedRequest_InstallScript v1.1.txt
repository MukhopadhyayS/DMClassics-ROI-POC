Use cabinet
go
--Create Multi links for HPF application process

Declare @queue varchar(50)

--System:  Multi Links

Select @Queue = 'SYSTEM: Document Trigger ROI'
If not exists (Select 1 from wq_temp_definition where queue = @queue)
	Begin
	Insert into wq_temp_definition (queue, type, result,from_date, to_date, CYCLE,  QUANTITY, FOLLOWUP, FU_TYPE, SCHEDULE, HOWFAST, RETAIN, SELECTABLE, 	EVALUATED,action_id, del_follow,doctype_id,chk_discharge,ASSIGNMENT_COUNT, mdp_ind,ActivityInstanceId, AgentID, DomainInstanceID, UserInstanceID)
	select @queue, 6,0, '1900-01-01', '1900-01-01',0,0,0,0,0,0,0,1, '1900-01-01', 99,0,0,0,0,0, (select nextid + 1 from wfe_nextvalue where name = 	'ACTIVITY'), 0,99,0
		
	Update wfe_nextvalue set nextid = (select max(activityinstanceid) from wq_temp_definition) where name = 'ACTIVITY'
	End

--System: Decision Objects


Select @Queue = 'SYSTEM: ROI Automated Process'
If not exists (Select 1 from wq_temp_definition where queue = @queue)
	Begin
	Insert into wq_temp_definition (queue, type, result,from_date, to_date, CYCLE,  QUANTITY, FOLLOWUP, FU_TYPE, SCHEDULE, HOWFAST, RETAIN, SELECTABLE, 	EVALUATED,action_id, del_follow,doctype_id,chk_discharge,ASSIGNMENT_COUNT, mdp_ind,ActivityInstanceId, AgentID, DomainInstanceID, UserInstanceID)
	select @queue,11,0, '1900-01-01', '1900-01-01',0,0,0,0,0,0,0,1, '1900-01-01', 99,0,0,0,0,0, (select nextid + 1 from wfe_nextvalue where name = 	'ACTIVITY'), 0,99,0
		
	Update wfe_nextvalue set nextid = (select max(activityinstanceid) from wq_temp_definition) where name = 'ACTIVITY'
	End


--System: Group Work Queues


Select @Queue = 'SYSTEM: ROI Automated Process Dead End'
If not exists (Select 1 from wq_temp_definition where queue = @queue)
	Begin
	Insert into wq_temp_definition (queue, type, result,from_date, to_date, CYCLE,  QUANTITY, FOLLOWUP, FU_TYPE, SCHEDULE, HOWFAST, RETAIN, SELECTABLE, 	EVALUATED,action_id, del_follow,doctype_id,chk_discharge,ASSIGNMENT_COUNT, mdp_ind,ActivityInstanceId, AgentID, DomainInstanceID, UserInstanceID)
	select @queue,18,1, '1900-01-01', '1900-01-01',0,0,0,0,0,0,1,1, '1900-01-01', 99,0,0,0,0,0, (select nextid + 1 from wfe_nextvalue where name = 	'ACTIVITY'), 0,99,0
		
	Update wfe_nextvalue set nextid = (select max(activityinstanceid) from wq_temp_definition) where name = 'ACTIVITY'
	End



--Create the sql fragments for the decision objects


If not exists (Select 1 from sql_fragments where DESCRIPTION = 'SP ROIAutomatedProcess')
	Insert into SQL_fragments values('SP ROIAutomatedProcess', 'Cabinet', 98, '', 'WFE_ROI_ProcessAutomatedRequest',1)



--Create the linkage for the work thread


GO
declare @queue varchar(50), @Test varchar(30), @resulting varchar(50)

Select @queue = 'SYSTEM: Document Trigger ROI', @test = '', @resulting = 'SYSTEM: ROI Automated Process'
If not exists (Select 1 from wq_temp_linkage where queue = @queue and test = @test and Resulting = @resulting)
	Insert into wq_temp_linkage values (@queue, @test, @resulting, getdate(),getdate(), 0)

Select @queue = 'SYSTEM: ROI Automated Process', @test = 'SP ROIAutomatedProcess', @resulting = 'SYSTEM: ROI Automated Process Dead End'

If not exists (Select 1 from wq_temp_linkage where queue = @queue and test = @test and Resulting = @resulting)
	Insert into wq_temp_linkage values (@queue, @test, @resulting, getdate(),getdate(), 0)
